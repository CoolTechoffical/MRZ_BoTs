LLama Curl Command

Python Code

import requests

# Define the API endpoint
url = "https://horrid-api.onrender.com/llama"

# Define the query parameter
query = {"query": "Hello"}

# Set the headers
headers = {"accept": "application/json"}

# Make the GET request
response = requests.get(url, headers=headers, params=query)

# Check if the request was successful
if response.status_code == 200:
    # Print the JSON response
    print(response.json())
else:
    # Print the error status code and message
    print(f"Error: {response.status_code}")
    print(response.text)

Palm Curl Command

Python Code

import requests
from requests.exceptions import ConnectionError, Timeout, RequestException

# Define the base URL and query parameter
base_url = "https://horrid-api.onrender.com/palm"
query = "Hey"
full_url = f"{base_url}?query={query}"

# Set the headers
headers = {"accept": "application/json"}

try:
    # Make the GET request
    response = requests.get(full_url, headers=headers, timeout=10)
    
    # Check if the request was successful
    response.raise_for_status()  # This will raise an HTTPError for bad responses

    # Parse the JSON response
    data = response.json()

    # Print the specific value from the JSON response
    result = data.get("result", "No result found")
    print(result)

except ConnectionError:
    print("Failed to establish a connection to the server. Please check your internet connection and the API URL.")
except Timeout:
    print("The request timed out. Please try again later.")
except RequestException as e:
    print(f"An error occurred: {e}")

Python Code for AI Models

import requests
from requests.exceptions import ConnectionError, Timeout, RequestException

# Define the base URL and query parameters
base_url = "https://horrid-api.onrender.com/ai"
query = "Hello"
model = 1
full_url = f"{base_url}?query={query}&model={model}"

# Set the headers
headers = {"accept": "application/json", "Content-Type": "application/json"}

# Define the role and query content
role_content = "assistant content here"
user_query = "user query here"

# Define the payload
payload = {
    "messages": [
        {
            "role": "assistant",
            "content": role_content
        },
        {
            "role": "user",
            "content": user_query
        }
    ]
}

try:
    # Make the POST request
    response = requests.post(full_url, headers=headers, json=payload, timeout=10)
    
    # Check if the request was successful
    response.raise_for_status()  # This will raise an HTTPError for bad responses

    # Parse the JSON response
    data = response.json()

    # Print the JSON response
    print(data)

except ConnectionError:
    print("Failed to establish a connection to the server. Please check your internet connection and the API URL.")
except Timeout:
    print("The request timed out. Please try again later.")
except RequestException as e:
    print(f"An error occurred: {e}")
